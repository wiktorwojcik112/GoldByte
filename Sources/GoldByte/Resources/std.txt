FN concat(left:STRING,right:STRING):STRING
	RETURN "%(left) %(right)"
/FN

FN input(text:STRING):STRING
	VAR STRING __temporary__ ""
	PRINT text
	INPUT STRING $__temporary__ true
	RETURN __temporary__
/FN

NAMESPACE math
	FN power(number:NUMBER,power:NUMBER):NUMBER
		IF # power == 0 #
			RETURN 1
		/IF

		IF # power == 1 #
			RETURN number
		/IF

		VAR NUMBER result 0
		ASSIGN $result number

		VAR NUMBER i 1

		WHILE # i != power #
			ASSIGN $result <result * number>
			ASSIGN $i <i + 1>
		/WHILE

		RETURN result
	/FN
	
	FN random(min:NUMBER,max:NUMBER):NUMBER
		VAR NUMBER __temporary__ 0
		RAND $__temporary__ min max
		RETURN __temporary__
	/FN
/NAMESPACE
